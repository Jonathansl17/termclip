#This is the bash configuration file for my custom scripts
export PATH="$HOME/bin:$PATH"

#Copy a file to clipboard from terminal
c() {
    if [ $# -lt 1 ]; then
        echo "Usage: c file1 file2 ..."
        return 1
    fi

    # Kill previous c.py processes
    pkill -f "$HOME/bin/c.py" 2>/dev/null

    # Launch new
    "$HOME/bin/c.py" "$@" >/dev/null 2>&1 & disown

    echo "Files copied to clipboard:"
    for f in "$@"; do
        echo "   $f"
    done
}

# Copy the text content of a file to the clipboard
cc() {
    if [ $# -lt 1 ]; then
        echo "Usage: cc file"
        return 1
    fi

    # Kill previous cc.py processes
    pkill -f "$HOME/bin/cc.py" 2>/dev/null

    # Launch new
    "$HOME/bin/cc.py" "$1" >/dev/null 2>&1 & disown

    echo "Content copied to clipboard from:"
    echo "   $1"
}

# Copy the current working directory path to the clipboard
cpwd() {
    target="${1:-$(pwd)}"

    # Kill previous cpwd.py processes
    pkill -f "$HOME/bin/cpwd.py" 2>/dev/null

    # Launch new
    "$HOME/bin/cpwd.py" "$target" >/dev/null 2>&1 & disown

    echo "Path copied to clipboard:"
    echo "   $target"
}

# Paste a file from the clipboard into the current directory
v() {
    output=$("$HOME/bin/v.py" "$@")
    if [ $? -ne 0 ]; then
        echo "No files found in clipboard."
        return 1
    fi

    echo "Files pasted from clipboard:"
    for f in $output; do
        echo "   $f"
    done
}
